
//Create DB
CREATE DATABASE TLA;

//Create Table
CREATE TABLE tlausers(
    uID INT(10) AUTO_INCREMENT PRIMARY KEY,
    NAME VARCHAR(255) NOT NULL,
    company VARCHAR(255) NOT NULL,
    email VARCHAR(255) NOT NULL,
    token VARCHAR(255) NOT NULL
);

CREATE TABLE trucks(
    tID INT(10) AUTO_INCREMENT PRIMARY KEY,
    tComp VARCHAR(255) NOT NULL,
    tMake VARCHAR(255) NOT NULL,
    tModel VARCHAR(255) NOT NULL,
    tMile VARCHAR(255) NOT NULL
);
